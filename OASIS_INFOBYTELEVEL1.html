)

# Normalize common names (robust to variations like "annual_income_(k$)")
def find_col(candidates, substrings):
    for c in candidates:
        for s in substrings:
            if s in c:
                return c
    return None

cols = list(df.columns)
rename_map = {}
cust = find_col(cols, ["customer", "id"])
if cust and cust != "customer_id": rename_map[cust] = "customer_id"
gen = find_col(cols, ["gender"])
if gen and gen != "gender": rename_map[gen] = "gender"
age = find_col(cols, ["age"])
if age and age != "age": rename_map[age] = "age"
inc = find_col(cols, ["annual_income", "income"])
if inc and inc != "annual_income": rename_map[inc] = "annual_income"
score = find_col(cols, ["spending_score", "spending", "score"])
if score and score != "spending_score": rename_map[score] = "spending_score"
df = df.rename(columns=rename_map)

# 6. HANDLE OUTLIERS
def remove_outliers_iqr(dataframe, column):
    # Convert to numeric (remove non-numeric entries)
    dataframe[column] = pd.to_numeric(dataframe[column], errors='coerce')
    
    # Drop NaNs for calculation
    Q1 = dataframe[column].quantile(0.25)
    Q3 = dataframe[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    return dataframe[(dataframe[column] >= lower_bound) & (dataframe[column] <= upper_bound)]
